def self.parse(feed_string, parser_type="RSS")
  if parser_type == "RSS"
    rss_feed = RSS::Parser.parse(feed_string)
    Feed.new(rss_feed.channel.title, rss_feed.channel.description, rss_feed.channel.link, rss_feed.channel.items)
  end
end
