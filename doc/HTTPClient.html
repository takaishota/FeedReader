def sendRequest(request_url)
  url = URI.parse(request_url)
  req = Net::HTTP::Get.new(url.path)

  begin
    res = Net::HTTP.start(url.host, url.port) do |http|
      http.open_timeout = 5
      http.read_timeout = 10
      http.request(req)
    end

    case res
    when Net::HTTPSuccess
      return res.body
    else
      # エラーハンドリング
      puts [url.to_s, e.class, e].join(" : ")
    end
  rescue => e
    # エラーハンドリング
    puts [url.to_s, e.class, e].join(" : ")
  end
end
